1.年龄段在“18-24” 的男性年轻人， 最喜欢看哪10部

package homework.chapter2

import org.apache.spark.sql.SparkSession

import scala.collection.mutable.ArrayBuffer


/**
 *
 * @author smallheroic
 * @Date 2019-12-11 19:49
 *
 */
object homework03 {

  def main(args: Array[String]): Unit = {

    val spark = SparkSession.builder().appName("homework03").master("local[4]").getOrCreate()
    val sc = spark.sparkContext

    val user_data=sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\users.dat").flatMap(_.split("\t"))

    val user_id = user_data.filter{line=>
      var tmp = (line.split("::")(2).toInt>=18) && (line.split("::")(2).toInt<=24)  && (line.split("::")(1)=="M")
      tmp
    }.map{line=>
      (line.split("::")(0),line.split("::")(0))}.collectAsMap()//(userid,userid)

    val user_bc = sc.broadcast(user_id)

    val rate_data = sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\ratings.dat").flatMap(_.split("\t")).map(line=>
      (line.split("::")(0),line.split("::")(1)))//(uid,mid)

    val movie_data = sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\movies.dat").flatMap(_.split("\t")).map(line=>
      (line.split("::")(0),line.split("::")(1)))//(mid,title)

    val res_mid = rate_data.mapPartitions(iter=>{
      val userId = user_bc.value
      val arr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(uid,mid)=>{
        if(userId.contains(uid)){
          arr.+=((mid,1))
        }
      }}
      arr.iterator
    }).reduceByKey(_+_).top(10)(Ordering.by(_._2)).toMap//(mid,count)

    val resid_bc = sc.broadcast(res_mid)

    val res = movie_data.mapPartitions(iter=>{
      val moveisId = resid_bc.value
      val resArr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(mid1,title)=>{
        if(moveisId.contains(mid1)){
          resArr.+=((title,moveisId.getOrElse(mid1,0)))
        }
      }}
      resArr.iterator
    })
    res.top(10)(Ordering.by(_._2)).foreach(line=>
      println("年龄段在“18-24” 的男性看多最多的10部电影:"+line._1+","+s"${line._2}位看过"))//(title ,count),只输出title
  }

}

2.得分最高的10部电影； 看过电影最多的前10个人；女性看多最多的10部电影；男性看过最多
的10部电影

package homework.chapter2

import org.apache.spark.sql.SparkSession

import scala.collection.mutable.ArrayBuffer

/**
 *
 * @author smallheroic
 * @Date 2019-12-12 16:45
 *
 */
object homework04 {

  val spark = SparkSession.builder().appName("homework04").master("local[4]").getOrCreate()
  val sc = spark.sparkContext

  val user_data = sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\users.dat").flatMap(_.split("\t")).map(line =>
    (line.split("::")(0), line.split("::")(1)))//(uid,gender)

  val rate_data = sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\ratings.dat").flatMap(_.split("\t")).map(line =>
    (line.split("::")(0), line.split("::")(1), line.split("::")(2).toDouble)) //(uid,mid,rate)

  val movie_data = sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\sparkcore\\data\\ml-1m\\movies.dat").flatMap(_.split("\t")).map(line =>
    (line.split("::")(0), line.split("::")(1))) //(mid,title)

  def ans1()={

    val rate_data_1 = rate_data.map(line=>(line._2,line._3)).groupByKey()//(mid,(rate1,rate2...))


    val res_mid = rate_data_1.map(info => {
      var sum: Double = 0
      var count: Double = 0
      var avg: Double = 0
      info._2.foreach { case rates => {
        sum += rates
        count += 1
      }
      }
      avg = sum / count
      (info._1, avg)
    }
    ).top(10)(Ordering.by(_._2)).toMap //(mid,rate_avg)
    res_mid.foreach(println)
    val resmid_bc = sc.broadcast(res_mid)
    val res = movie_data.mapPartitions(iter => {
      val movieId = resmid_bc.value
      val resMIdArr = ArrayBuffer[(String, Double)]()
      iter.foreach { case (mid1, title) => {
        if (movieId.contains(mid1))
          resMIdArr.+=((title, movieId.getOrElse(mid1, 0)))
      }
      }
      resMIdArr.iterator
    })
    res.top(10)(Ordering.by(_._2)) //(title ,avg),只输出title
  }

  def ans2()={
    val rate_data_2 = rate_data.map(line=>(line._1,1))   //(uid,1)
    rate_data_2.reduceByKey(_+_).top(10)(Ordering.by(_._2)) //(uid,count)
  }

  def ans3()={
    val user_data_3  = user_data.filter(_._2=="F").collectAsMap()//(uid,F)
    val userId_3_bc = sc.broadcast(user_data_3)
    val resMId_3  = rate_data.map(line=>(line._1,line._2)).mapPartitions(iter=>{
      val uid_3 = userId_3_bc.value
      val resMId_3Arr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(uid,mid)=>{
        if(uid_3.contains(uid))
          resMId_3Arr.+=((mid,1))
      }}
      resMId_3Arr.iterator
    }).reduceByKey(_+_).top(10)(Ordering.by(_._2)).toMap//(mid,count)

    val resMTitle_3bc = sc.broadcast(resMId_3)

    val res = movie_data.mapPartitions(iter=>{
      val moveisId = resMTitle_3bc.value
      val resArr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(mid1,title)=>{
        if(moveisId.contains(mid1)){
          resArr.+=((title,moveisId.getOrElse(mid1,0)))
        }
      }}
      resArr.iterator
    })
    res.top(10)(Ordering.by(_._2))//(title ,count)
  }

  def ans4()={
    val user_data_3  = user_data.filter(_._2=="M").collectAsMap()//(uid,F)
    val userId_3_bc = sc.broadcast(user_data_3)
    val resMId_3  = rate_data.map(line=>(line._1,line._2)).mapPartitions(iter=>{
      val uid_3 = userId_3_bc.value
      val resMId_3Arr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(uid,mid)=>{
        if(uid_3.contains(uid))
          resMId_3Arr.+=((mid,1))
      }}
      resMId_3Arr.iterator
    }).reduceByKey(_+_).top(10)(Ordering.by(_._2)).toMap//(mid,count)

    val resMTitle_3bc = sc.broadcast(resMId_3)

    val res = movie_data.mapPartitions(iter=>{
      val moveisId = resMTitle_3bc.value
      val resArr = ArrayBuffer[(String,Int)]()
      iter.foreach{case(mid1,title)=>{
        if(moveisId.contains(mid1)){
          resArr.+=((title,moveisId.getOrElse(mid1,0)))
        }
      }}
      resArr.iterator
    })
    res.top(10)(Ordering.by(_._2))//(title ,count)
  }


  def main(args: Array[String]): Unit = {
    //val start = System.nanoTime()
    ans1().foreach { line =>
      println("得分最高的10部电影:" + line._1+","+s"平均得分为：${line._2}")
    }//只输出title
    ans2().foreach{line=>
      println("看过电影最多的前10个人:"+line._1+","+s"看过${line._2}部电影")
    }
    ans3().foreach{line=>
      println("女性看多最多的10部电影:"+line._1+","+s"${line._2}位女性看过")
    }
    ans4().foreach{line=>
      println("男性看多最多的10部电影:"+line._1+","+s"${line._2}位男性看过")
    }
    //val end = System.nanoTime()
   // println("------------------------------------")
  //  println("程序运行时间:[" + (end - start) + "ns]")
  //  println("------------------------------------")
  }

}

3.利用数据集SogouQ2012.mini.tar.gz 将数据按照访问次数进行排序， 求访问量前10的网站

import org.apache.spark.sql.SparkSession

/**
 *
 * @author smallheroic
 * @Date 2019-12-12 21:12
 *
 */
object homework05 {

  def main(args: Array[String]): Unit = {

    val spark = SparkSession.builder().appName("homework05").master("local[4]").getOrCreate()
    val sc = spark.sparkContext

    sc.textFile("E:\\FTP\\spark\\3-sparkCore2\\SogouQ2012.mini.tar.gz")
      .map(line=>(line.split("\t").reverse(0),1))

    println("访问量前10的网站:"+line._1+","+s"访问量为${line._2}"))
